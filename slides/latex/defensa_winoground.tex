\section{El problema}

% ===================== 3. LOS DOS PAPERS =====================
\begin{frame}{Mi tema: dos papers que cuestionan a CLIP}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Winoground} (Thrush et al., 2022)\\
    \footnotesize ¿Los modelos visión-lenguaje razonan de forma \emph{composicional},
    o reconocen objetos sueltos? Mide el orden y la vinculación, no el reconocimiento.
    \column{0.5\textwidth}
    \textbf{Evaluating CLIP} (Agarwal et al., 2021)\\
    \footnotesize Una accuracy mayor no define un modelo «mejor»: el diseño de clases y
    los prompts esconden sesgos. La métrica no lo es todo.
  \end{columns}
  \vfill
  \centering\footnotesize Ambos atacan la misma idea: \textbf{una buena métrica puede ocultar que el modelo no entiende.}
\end{frame}

% ===================== 4. QUÉ ES UN PAR MÍNIMO =====================
\begin{frame}{El núcleo de Winoground: el par mínimo}
  Dos imágenes y dos captions con \textbf{exactamente las mismas palabras en distinto orden}.
  \begin{center}
  \begin{tabular}{ll}
    \toprule
    Caption 0 & «\textit{an old person kisses a young person}» \\
    Caption 1 & «\textit{a young person kisses an old person}» \\
    \bottomrule
  \end{tabular}
  \end{center}
  \footnotesize Mismo vocabulario. Distinguirlas exige saber \emph{quién} hace \emph{qué} a \emph{quién}.
  Un modelo que solo detecta «old», «young», «kiss» no puede; necesita composición.
  \par\vspace{0.5em}
  400 ejemplos, etiquetados por tipo de cambio: \textbf{Object}, \textbf{Relation}, \textbf{Both}.
\end{frame}

% ===================== 5. LAS TRES MÉTRICAS =====================
\begin{frame}{Las tres métricas oficiales}
  Por ejemplo, similitud $s(\text{caption}, \text{imagen})$ en una matriz $2\times2$:
  \begin{columns}[T]
    \column{0.46\textwidth}
    \centering
    \begin{tabular}{c|cc}
        & $i_0$ & $i_1$ \\\hline
      $c_0$ & \textbf{\color{verde}$s_{00}$} & $s_{01}$ \\
      $c_1$ & $s_{10}$ & \textbf{\color{verde}$s_{11}$} \\
    \end{tabular}
    \par\footnotesize (la diagonal verde es lo correcto)
    \column{0.54\textwidth}
    \footnotesize
    \textbf{text score}: fijada cada imagen, ¿gana el caption correcto?\\
    $s_{00}>s_{10}$ y $s_{11}>s_{01}$\\[4pt]
    \textbf{image score}: fijado cada caption, ¿gana la imagen correcta?\\
    $s_{00}>s_{01}$ y $s_{11}>s_{10}$\\[4pt]
    \textbf{group score}: ambas a la vez (text \textbf{y} image).
  \end{columns}
\end{frame}

% ===================== 6. EL AZAR NO ES 1/16 =====================
\begin{frame}{Detalle fino: el azar del group score es 1/6}
  \begin{itemize}
    \item text e image \textbf{no son independientes}: comparten las 4 similitudes.
    \item group = 1 exige que \emph{ambas} diagonales superen a \emph{ambas} off-diagonales.
    \item De las $4! = 24$ ordenaciones de 4 valores, solo \textbf{4} cumplen $\Rightarrow 4/24 = 1/6 \approx 0.167$.
  \end{itemize}
  \vfill
  \centering\alert{No es $1/16$.} Comparar contra el baseline correcto es parte de la métrica
  (un error común que un examinador puede preguntar).
\end{frame}

% ===================== 7. LA TESIS =====================
\begin{frame}{La pregunta que respondo}
  \Large
  \centering
  ¿Un buen \textbf{Recall@K} (recuperación) implica
  \textbf{razonamiento composicional}?
  \vfill
  \normalsize Voy a mostrar que \textbf{no}: el mismo modelo, sobre el mismo conjunto,
  recupera bien y \alert{falla la composición}.
\end{frame}

\section{Cómo lo medí (Cuaderno 10)}

% ===================== 8. CÓMO FUNCIONA CLIP =====================
\begin{frame}{Cómo funciona CLIP: un dual-encoder contrastivo}
  \begin{itemize}
    \item Dos torres separadas: una codifica la imagen, otra el texto.
    \item Cada modalidad colapsa en \textbf{un único vector global}; se comparan por \textbf{coseno}.
    \item Entrenado con aprendizaje contrastivo (acercar pares correctos, alejar incorrectos).
  \end{itemize}
  \vfill
  \footnotesize Esto es exactamente el motor del \textbf{Cuaderno 10} (OpenCLIP):
  \texttt{create\_model\_and\_transforms}, \texttt{encode\_image/encode\_text}, similitud coseno.
  Lo reutilicé tal cual.
\end{frame}

% ===================== 9. POR QUÉ FALLA (intuición) =====================
\begin{frame}{Por qué un dual-encoder falla la composición}
  \begin{columns}[T]
    \column{0.6\textwidth}
    \begin{itemize}
      \item Un vector global se comporta como \textbf{«bolsa de conceptos»}.
      \item «old», «young», «kiss» activan dimensiones parecidas \emph{estén como estén compuestos}.
      \item No hay interacción \emph{token-a-token} entre palabras y regiones.
    \end{itemize}
    \column{0.4\textwidth}
    \footnotesize\centering
    \textbf{Conexión C5 / C8:}\\[4pt]
    La \emph{fusión profunda} (MMBT, C5) y la \emph{atención crossmodal} (C8)
    sí dejan interactuar tokens visuales y textuales. Esa es la vía para la composición.
  \end{columns}
\end{frame}

% ===================== 10. PIPELINE =====================
\begin{frame}{El pipeline (reproducible)}
  \footnotesize
  \begin{enumerate}
    \item \texttt{winoground\_data.py}: Winoground real (HF, gated) o set curado offline.
    \item \texttt{openclip\_utils.py}: embeddings L2-normalizados (motor C10).
    \item \texttt{winoground\_eval.py}: matriz $2\times2$ y los 3 scores.
    \item \texttt{metrics.py}: Recall@K, intervalos bootstrap.
    \item \texttt{blindness\_probe.py}: ¿usa la imagen? · \texttt{error\_analysis.py}: por tag.
  \end{enumerate}
  \vfill
  Todo determinista (semillas), versionado y con tests. Una corrida: \texttt{make run}.
\end{frame}

% ===================== 11. EL SCORER (CÓDIGO) =====================
\begin{frame}[fragile]{El bloque clave: el scorer}
\begin{verbatim}
def text_correct(sim):   # fija imagen, elige caption
    return sim[0,0] > sim[1,0] and sim[1,1] > sim[0,1]

def image_correct(sim):  # fija caption, elige imagen
    return sim[0,0] > sim[0,1] and sim[1,1] > sim[1,0]

def group_correct(sim):
    return text_correct(sim) and image_correct(sim)
\end{verbatim}
  \footnotesize Es el código \textbf{oficial} de Winoground. La redacción verbal de papers
  de seguimiento suena al revés; me guié por el código y lo validé numéricamente.
\end{frame}

% ===================== 14. LEER EL RESULTADO =====================
\begin{frame}{Cómo leer ese resultado}
  \begin{itemize}
    \item \textbf{text} (0.35) está \emph{por encima} del azar (0.25): el modelo capta algo.
    \item \textbf{group} (0.075) está \emph{por debajo} del azar (0.167).
    \item No significa «peor que aleatorio» en general. Es la \textbf{asimetría text $\gg$ image}:
    acierta una dirección y falla la otra, y el group exige ganar las dos.
  \end{itemize}
  \vfill
  \centering El intervalo de confianza del group (bootstrap 95\%: $[0.05, 0.10]$) queda
  \textbf{entero por debajo} del azar.
\end{frame}

% ===================== 22. LIMITACIONES =====================
\begin{frame}{Defensa crítica: limitaciones}
  \begin{itemize}
    \item El group bajo \textbf{mezcla} fallo composicional con ítems ambiguos o difíciles
    (Diwan et al., «Why is Winoground Hard?»).
    \item Las métricas son comparaciones estrictas: \textbf{sensibles a empates} y a la normalización.
    \item R@K y accuracy \textbf{no miden composición}: necesarias, no suficientes.
    \item El set curado offline es un \emph{proxy} sintético (OOD para CLIP).
  \end{itemize}
\end{frame}

% ===================== 23. MEJORAS =====================
\begin{frame}{Mejora propuesta}
  \begin{itemize}
    \item Evaluar un \textbf{cross-encoder / fusión profunda} (C5) y \textbf{re-ranking} con
    interacción cruzada (C6): la arquitectura que sí puede componer.
    \item Versionar embeddings (DVC) para reproducibilidad total de datos.
    \item Ampliar a más prompts, checkpoints y a \emph{prompt ensembles} (C10).
  \end{itemize}
  \vfill
  \centering\footnotesize Lo que \textbf{no} aseguro aún: que un cross-encoder lo resuelva. Es mi hipótesis, no un resultado medido.
\end{frame}
